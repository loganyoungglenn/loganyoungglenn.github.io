bigmbb <- logs |> 
  filter(Team %in% sweet)|>
  select(Team, TeamFG, TeamFGA, Team3P, Team3PA) |>
  group_by(Team) |>
  summarize(
    TotalFG = sum(TeamFG),
    TotalFGA = sum(TeamFGA),
    Total3P = sum(Team3P),
    Total3PA = sum(Team3PA)
  )|>
  mutate(FGPct = TotalFG/TotalFGA, ThreePct = Total3P/Total3PA) |>
  select(Team, Total3P, Total3PA, ThreePct) |>
  arrange(desc(ThreePct))
